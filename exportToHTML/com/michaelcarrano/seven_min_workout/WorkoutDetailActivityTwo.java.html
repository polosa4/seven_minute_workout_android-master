14   /** 
15    * An activity representing a single Workout detail screen. This activity is only used on handset 
16    * devices. On tablet-size devices, item details are presented side-by-side with a list of items in 
17    * a {@link WorkoutListActivity}. <p> This activity is mostly just a 'shell' activity containing 
18    * nothing more than a {@link WorkoutDetailFragment}. 
19    */ 
20   public class WorkoutDetailActivityTwo extends BaseActivity { 
21    
22       @Override 
23       protected void onCreate(Bundle savedInstanceState) { 
24           super.onCreate(savedInstanceState); 
25           setContentView(R.layout.activity_workout_detail); 
26    
27           // Get the Workout that was selected 
28           int selected = getIntent().getExtras().getInt(WorkoutDetailFragment.ARG_WORKOUT_POS); 
29           WorkoutContent.Workout workout = WorkoutContent.WORKOUTS.get(selected); 
30           Log.i("7min", "Selected: " + selected + " " + workout.name); 
31    
32           toolbar.setTitle(workout.name); 
33           toolbar.setBackgroundColor(workout.light); 
34           if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.LOLLIPOP) { 
35               Window window = getWindow(); 
36               window.addFlags(WindowManager.LayoutParams.FLAG_DRAWS_SYSTEM_BAR_BACKGROUNDS); 
37               window.setStatusBarColor(workout.dark); 
38           } 
39    
40           // savedInstanceState is non-null when there is fragment state 
41           // saved from previous configurations of this activity 
42           // (e.g. when rotating the screen from portrait to landscape). 
43           // In this case, the fragment will automatically be re-added 
44           // to its container so we don't need to manually add it. 
45           // For more information, see the Fragments API guide at: 
46           // 
47           // http://developer.android.com/guide/components/fragments.html 
48           // 
49           if (savedInstanceState == null) { 
50               // Create the detail fragment and add it to the activity 
51               // using a fragment transaction. 
52               Bundle arguments = new Bundle(); 
53    
54               arguments.putInt(WorkoutDetailFragmentTwo.ARG_WORKOUT_POS, selected); 
55               WorkoutDetailFragmentTwo fragment = new WorkoutDetailFragmentTwo(); 
56               fragment.setArguments(arguments); 
57               getSupportFragmentManager().beginTransaction().add(R.id.workout_detail_container, 
58                       fragment) 
59                       .commit(); 
60           } 
61       } 
62    
63       @Override 
64       public boolean onOptionsItemSelected(MenuItem item) { 
65           int id = item.getItemId(); 
66           if (id == android.R.id.home) { 
67               // This ID represents the Home or Up button. In the case of this 
68               // activity, the Up button is shown. Use NavUtils to allow users 
69               // to navigate up one level in the application structure. For 
70               // more details, see the Navigation pattern on Android Design: 
71               // 
72               // http://developer.android.com/design/patterns/navigation.html#up-vs-back 
73               // 
74               NavUtils.navigateUpTo(this, new Intent(this, WorkoutListActivity.class)); 
75               return true; 
76           } 
77           return super.onOptionsItemSelected(item); 
78       }
